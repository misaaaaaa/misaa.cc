                    #include <LedControl.h>
                    //Definir número de matrices
                    #define NumMatrix 1

                    //Definir pines de MAX7219
                    // LedControl(DIN,CLOCK,CS, Número de matrices)
                    LedControl lc=LedControl(12,11,10, NumMatrix);

                    void setup() {
                      // Inicializar matriz

                      lc.shutdown(0,false);  // Activar la matriz
                      lc.setIntensity(0,8);  // Poner el brillo a un valor intermedio
                      lc.clearDisplay(0);    // Y borrar todo

                    }

                    void loop() {
                      // lc.setLed(Nº MATRIZ, FILA, COLUMNA, ESTADO);

                      for (int j = 0; j<=7 ; j++){
                        for (int i = 0; i<=7 ; i++){
                          lc.setLed(0, i, j, HIGH);
                          delay(100);
                          lc.setLed(0, i, j, LOW);
                        }
                      }

                      for (int j = 7; j>=0 ; j--){
                        for (int i = 7; i>=0 ; i--){
                          lc.setLed(0, i, j, HIGH);
                          delay(100);
                          lc.setLed(0, i, j, LOW);
                        }
                      }
                    }

      20210712 | proj02

      ↑
    • Generador de clock de subdivisiones por divisiones binarias y ternarias: 1, 2, 3, 4, 6, 8, 9, 12, 16
    • Activación de trigger en 9 salidas
    • Inclusión de corrimiento de cada pulso aleatorizado
    • Diseño circular, similar a ritmos euclideanos
    • Diagrama de temporización, con mínimo común múltiplo 144 para las subdivisiones dadas (última fila representa la sumatoria ∑, o compuerta OR de todos los clocks previos)
    • Para visualizar, se utilizará el integrado MAX7219, que permite encender hasta 64 leds de forma independiente, y de muy bajo consumo. Funciona a través de la librería LedControl, y requiere 3 pines del arduino: Digital IN, Clock y Latch (Load)
    • El mapeo por paso será realizado según la siguiente matriz. Se indican los pines correspondientes del integrado.
    • Corrimiento de pulso (random range): Inclusión de perilla para randomizar la posición de un trigger según rango dado. Debe incluirse en valores posibles su relación al tamaño de espacios disponible hacia atrás y delante.
    • El diseño del algoritmo debe considerar un tamaño de muestreo múltiplo de 144
    • Es posible que sea necesario disminuir el número de salidas para efectos de este prototipo

                    if (indice % (samples / divs[i]) == 0) { //SUBDIVISIONES POSIBLES

                      flag[i] = 0;
                      pFlag[i] = flag[i];
                    }

                    // Chequeo posiciones Leds
                    posX[i] = initX[i] + index[i];
                    posY[i] = initY[i];
                    if (posX[i] > 7) {
                      posX[i] = index[i] - (8 - initX[i]);
                      posY[i] = initY[i] + 1;
                    }

                    //Encendido Leds
                    if (ledState[i] == HIGH) {
                      lc.setLed(0, posX[i], posY[i], HIGH);
                    }
                    if ((currentMillis - ledMillis[i]) > onTime && ledState[i] == HIGH) {
                      lc.setLed(0, posX[i], posY[i], LOW);
                      ledState[i] = LOW;
                    }

                    // Encendido Outputs
                    if ((currentMillis - ledMillis[i]) > outTime && outState[i] == HIGH) {
                      digitalWrite(out[i], LOW);
                      outState[i] = LOW;
                    }

      • 20210810 | proj07

      ↑
    • Video explicación prototipo, conectando a un módulo de Relé (5V), para escuchar su cambio de estado, y explorando sus relaciones rítmicas:
    • Descripción: De comportamiento similar a múltiples metrónomos sincronizados, Resorritmia es un generador de señales de clock de 9 subdivisiones [1, 2, 3, 4, 6, 8, 9, 12, 16], las cuales pueden ser utilizadas de forma simultánea. Esto puede ser usado de base para el diseño de futuras máquinas de estado (State Machines, s. f.), aparatos capaces de recibir señales y realizar instrucciones sencillar (contar, sumar, mover, sonar o prender). Al poseer un reloj común, pueden generar sincronías rítmicas entre los elementos que se conecten. Un comportamiento habitual en circuitos integrados como el CD4040BE, que divide en potencias de 2. Este módulo además incorpora divisiones en 3, permitiendo rítmicas ternarias, similares a las que se encuentran en la música latinoamericana (Miranda & Salinas, 2015).
    • Ejemplo de integrado CD4040
    • Referencias
      State Machines: Blink.ino learns to snooze—News—SparkFun Electronics. (s. f.). Recuperado 10 de agosto de 2021, de https://www.sparkfun.com/news/1801
      Miranda, S., & Salinas, H. (2015). Ritmos y tiempos de América: Entrevista con Horacio Salinas. Guaraguao, 19(50), 39-48. Retrieved August 10, 2021, from http://www.jstor.org/stable/44078657
    • Especificaciones:

      - 9 salidas independientes, con divisiones por 1, 2, 3, 4, 6, 8, 9, 12, 16; de triggers positivos de aproximadamente 30ms de duración
      - Control de velocidad a través de potenciómetro
      - Control de velocidad a través de señal externa (Control de Voltaje)
      - Botón de reset
      - Reset a través de señal externa (trigger positivo)
      - Pantalla de visualización basada en el circuito integrado MAX7219
      - Programada en Arduino Nano
